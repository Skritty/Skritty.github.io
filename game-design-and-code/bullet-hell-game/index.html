Bullet Hell Game

Made during fall 2016 college semester
By Trevor Thacker

This is a game I made for my final project of Intro Into Engineering and Computer Science. It is made in processing. The idea of the game is that you have to dodge large amounts of incoming bullets, and reflect some back at the enemies in order to kill them.

The player class includes stats such as health and movement speed, along with all the required input and collision functions to make movement work.

public class player extends Sprite{
  private int hp;
  private int cd;
  private int cdLeft;
  private float speed;
  private Sprite deflect;
  
  player(float x, float y, int c, float f, PApplet app, Sprite s){
    super(app,"player.jpg",0);
    setX(x);
    setY(y);
    hp=c;
    cd=600;
    cdLeft=0;
    speed=f;
    deflect = s;
    setCollisionRadius(20);
    setScale(.5);
    deflect.setScale(.4);
    deflect.setXY(x,y-height+25);
    deflect.setVisible(false);
  }
  void drawHitbox(){if(!isDead())ellipse((float)getX(), (float)getY(),(float)getCollisionRadius(), (float)getCollisionRadius());}
  int getHP(){return hp;}
  int getCD(){return cd;}
  int getCDLeft(){return cdLeft;}
  Sprite getDeflect(){return deflect;}
  float getPlayerSpeed(){return speed;}
  void setHP(int a){hp=a;}
  void setCD(int a){cd=a;}
  void setCDLeft(int a){cdLeft=a;}
  void setPlayerSpeed(float a){speed=a;}
  void movePlayer(){
      if(controls[5]&&p1.getCDLeft()<=0)deflect.setVisible(true);
      else deflect.setVisible(false);
      if(controls[0]&&!hitLeftWall()){
        if(!controls[4])setX(getX()-speed);
        if(controls[4])setX(getX()-speed/2);
      }
      if(controls[1]&&!hitRightWall()){
        if(!controls[4])setX(getX()+speed);
        if(controls[4])setX(getX()+speed/2);
      }
      if(controls[2]&&!hitTopWall()){
        if(!controls[4])setY(getY()-speed);
        if(controls[4])setY(getY()-speed/2);
      }
      if(controls[3]&&!hitBottomWall()){
        if(!controls[4])setY(getY()+speed);
        if(controls[4])setY(getY()+speed/2);
      }
      deflect.setXY(getX(),getY()-height+25);
  }
  boolean hitLeftWall(){
    if(getX()-getCollisionRadius()<=0){return true;}
    return false;
  }
  boolean hitRightWall(){
    if(getX()+getCollisionRadius()>=w){return true;}
    return false;
  }
  boolean hitTopWall(){
    if(getY()-getCollisionRadius()<0){return true;}
    return false;
  }
  boolean hitBottomWall(){
    if(getY()+getCollisionRadius()>=h){return true;}
    return false;
  }
  boolean hitWallSpecial(){
    return false;
  }
}
void keyPressed(){
  if(key==CODED){
    if(keyCode==LEFT)controls[0]=true;
    if(keyCode==RIGHT)controls[1]=true;
    if(keyCode==UP)controls[2]=true;
    if(keyCode==DOWN)controls[3]=true;
    if(keyCode==SHIFT)controls[4]=true;
    
  }
  if(key=='z'||key=='Z')controls[5]=true;
}
void keyReleased(){
  if(key==CODED){
    if(keyCode==LEFT)controls[0]=false;
    if(keyCode==RIGHT)controls[1]=false;
    if(keyCode==UP)controls[2]=false;
    if(keyCode==DOWN)controls[3]=false;
    if(keyCode==SHIFT)controls[4]=false;
  }
  if(key=='z'||key=='Z')controls[5]=false;
}
public class enemy extends Sprite{
  private int health;
  private int[] seq;
  private int stage;
  private int type;
  private boolean firstDone;
  private double angle = 0;
  private PApplet app;
  private ArrayList<bullet> bulletSpam = new ArrayList<bullet>();
  private ArrayList<bullet> toDelete = new ArrayList<bullet>();//get around errors
  private ArrayList<special> specials = new ArrayList<special>();
  enemy(float x, float y, int hp, int[] d, int t, PApplet a){
    super(a,"enemy.jpg",1);
    setXY(x,y);
    setScale(.5);
    health = hp;
    seq = d;
    firstDone = false;
    stage = 0;
    type = t;
    app = a;//idk what this is
    setCollisionRadius(75);
    switch(type){                                       //Enemy AI start
    case 0:{setVelX(random(-.004,.004));setAccXY(random(-.000001,.000001),random(0,.000001));}
      break;
    case 1:{setVelX(random(-.004,.004));setAccXY(random(-.000001,.000001),random(0,.000001));}
      break;
    }
    (new fireBullets()).start();
    (new moveBullets()).start();
    (new kill()).start();
    (new collision(this)).start();
  }
  void drawHitbox(){
    synchronized(bulletSpam){for(bullet b: bulletSpam){if(b.showHitbox())b.drawHitbox();}}
    if(!isDead())synchronized(this){ellipse((float)getX(), (float)getY(),(float)getCollisionRadius(), (float)getCollisionRadius());}
  }
  double getAngle(){return angle;}
  void setAngle(double x){angle = x;}
  int getHP(){return health;}
  boolean getFirstDone(){return firstDone;}
  int[] getSeq(){return seq;}
  void setHP(int a){health=a;}
  void setSeq(int[] a){seq=a;}
  ArrayList<bullet> getSpam(){synchronized(bulletSpam){return bulletSpam;}}
                      //-------------Enemy AI stuff--------------
  void moveAI(){
    switch(type){
    case 0:{}
    }
  }
  
  void fireBullet(){                                //Add the ability to fire many different types here
  if(!firstDone&&!bulletSpam.isEmpty())firstDone = true;
    if(seq.length>stage){
      switch(seq[stage]){
        case 0:{
          synchronized(bulletSpam){
            bulletSpam.add(new bullet(getX(),getY(),0,"bullet.jpg",app,this));
          }
          delay(600);
        }
          break;
        case 1:{
          synchronized(specials){
            specials.add(new special(getX(),getY(),0,"bullet.jpg",fire3,app,this));
          }
          delay(1200);
        }
          break;
      }
    }
    stage++;
  }
  class moveBullets extends Thread{
    private boolean limiter;
    private boolean stop = false;
    void run(){
      limiter = false;
      while(!stop){
        if(!limiter){
          synchronized(bulletSpam){
          for(bullet b: bulletSpam){
            if(!b.isDead())b.moveBullet();
          }}
          synchronized(specials){
          for(special s: specials){
            if(!s.isDead())s.moveBullet();
          }}
          limiter = true;
        }
        if(millis()%5==0&&millis()%20!=0)limiter = false;
        if(bulletSpam.isEmpty()&&millis()>2000){print("stopped updating bullets!\n");stop = true;}
      }
    }
  }
  class fireBullets extends Thread{
    private boolean limiter;
    private boolean alive;
    void run(){
      limiter = false;
      alive = true;
      while(alive){
        if(!limiter){
          fireBullet();
          limiter = true;
        }
        if(millis()%5==0&&millis()%20!=0)limiter = false;
        if(!isOnScreem()||isDead()){print("enemy stopped firing!\n");alive = false;}
      }
    }
  }
  class kill extends Thread{
    private boolean limiter;
    private boolean stop = false;
    void run(){
      limiter = false;
      while(!stop){
        if(!limiter){
          toDelete.clear();
          synchronized(bulletSpam){toDelete.addAll(bulletSpam);}
            for(bullet x: toDelete){
              if(!x.isOnScreem()){synchronized(bulletSpam){bulletSpam.remove(x);}}
            }
          limiter = true;
        }
        if(millis()%5==0&&millis()%20!=0)limiter = false;
        if(bulletSpam.isEmpty()&&millis()>2000){print("stopped killing bullets!\n");stop = true;}
      }
    }
  }
}

    setScale(.2);
    type=c;
    deflected = false;

    setCollisionRadius(15);

     }

  void fireBullet(){//Add the ability to fire many different types here
    if(seq.length>stage){
      switch(seq[stage]){
        case 0:{
          synchronized(firedBy.bulletSpam){
            firedBy.bulletSpam.add(new bullet(getX(),getY(),0,"bullet.jpg",app,this));

          limiter = true;
        }
        if(millis()%5==0&&millis()%20!=0)limiter = false;
        if(!isOnScreem()||isDead()||seq.length==stage){print("special stopped firing!\n");alive = false;}
        if(seq.length==stage){setDead(true);}
      }
    }

      while(!stop&&!p1.isDead()){
        if(!limiter){
          synchronized(e.bulletSpam){
          toDelete.clear();
          toDelete.addAll(e.bulletSpam);
          
          for(bullet b: toDelete){
            if(controls[5]==true&&p1.getCDLeft()<=0&&p1.getDeflect().isVisible()){
              if(p1.getDeflect().bb_collision(b)&&!b.isDeflected()){
                b.setDeflected(true);
                p1.setCDLeft(600);                                  //Cooldown of Deflect <--
              }
            }
            if(!b.isDead()){
              if(p1.cc_collision(b)){
                p1.setHP(p1.getHP()-1);

                b.setDead(true);
